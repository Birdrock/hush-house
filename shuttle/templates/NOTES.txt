  * Grab the Grafana credentials:

    User:
      kubectl get \
        --namespace {{ .Release.Namespace }} \
        secrets {{ .Release.Name }}-grafana \
        -o 'jsonpath={.data.admin-user}' | base64 -D

    Password:
      kubectl get \
        --namespace {{ .Release.Namespace }} \
        secrets {{ .Release.Name }}-grafana \
        -o 'jsonpath={.data.admin-password}' | base64 -D
